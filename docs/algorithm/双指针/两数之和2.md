/**
题目: 两数之和
输入: 数组和目标数 
输出: 两数相加等于目标值的可能性的集合, 每个数只能用一次
例子1:
输入: [1, 1, 2, 3, 5, 4], 6 
1 1 2 3 4 5
输出: [[1, 5], [2, 4]]
例子2:
输入: [1, 1, 2, 4, 3, 5, 5], 6
输出: [[1, 5], [1, 5], [2, 4]]
例子3:
输入: [1, 5, 5], 6
输出: [[1, 5]]
*/
function sumTarget(arr, target) {
    arr.sort((a, b) => a - b)
    let result = []
    let left = 0, right = arr.length - 1
    while (left < right) {
        let sum = arr[left] + arr[right]
        if (sum === target) {
            result.push([arr[left], arr[right]])
            left++
            right--
        } else if (sum < target) {
            left++
        } else {
            right--
        }
    }
    console.log(result)
}
sumTarget([1, 1, 2, 3, 5, 4], 6)
sumTarget([1, 1, 2, 4, 3, 5, 5], 6)
